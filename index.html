<script>
function filterTerms() {
const input = document.getElementById('search').value.toLowerCase();
const rows = document.querySelectorAll('#glossaryTable tbody tr');

rows.forEach(row => {
const text = row.textContent.toLowerCase();
row.style.display = text.includes(input) ? '' : 'none';
});
}

 function filterTerms() {
const input = document.getElementById('search').value.toLowerCase();
const rows = document.querySelectorAll('#glossaryTable tbody tr');

rows.forEach(row => {
const text = row.textContent.toLowerCase();
row.style.display = text.includes(input) ? '' : 'none';
});
}

function speak(text) {
const utterance = new SpeechSynthesisUtterance(text);
utterance.lang = 'en-US';
speechSynthesis.speak(utterance);
}

function addTerm() {
const term = document.getElementById('newTerm').value;
const defEn = document.getElementById('newDefEn').value;
const defEs = document.getElementById('newDefEs').value;

if (!term || !defEn || !defEs) {
alert('Por favor completa todos los campos');
return;
}

const table = document.querySelector('#glossaryTable tbody');
const row = table.insertRow();

row.insertCell(0).textContent = term;
row.insertCell(1).innerHTML = defEn + ' <button onclick="speak(\'' + defEn + '\')">🔊</button>';
row.insertCell(2).textContent = defEs;

document.getElementById('newTerm').value = '';
document.getElementById('newDefEn').value = '';
document.getElementById('newDefEs').value = '';
}
